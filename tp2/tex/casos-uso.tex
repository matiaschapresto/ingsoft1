  \ucpre{True}
  \ucpost{El sistema registra al nuevo pasajero}
  \ucactor{Pasajero}
\end{usecase}
\begin{usecasesteps}
  \ucstep{El pasajero ingresa sus datos(ver modelo conceptual).}{}

  \ucstep{Fin caso de uso.}{}
\end{usecasesteps}
