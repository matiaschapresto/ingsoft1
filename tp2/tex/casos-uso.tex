\subsection{Casos de Uso}

\diagramav{diag-cu}

\begin{usecase}
  \ucname{Registrandose en el Sistema}
  \ucpre{True}
  \ucpost{El sistema registra al nuevo usuario}
  \ucactor{Usuario}
\end{usecase}
\begin{usecasesteps}
  \ucstep{El usuario ingresa sus datos(ver modelo conceptual).}{}
  \ucstep{El sistema verifica que los datos sean validos y almacena al nuevo usuario.}
        {\ucalt{Si los datos son invalidos, el mostrar el mensaje de error correspondiente y volver a 1.}}
  \ucstep{Puede ser extendido por \texttt{CU Logueandose en el sistema}}{}
  \ucstep{Fin caso de uso.}{}
\end{usecasesteps}

\begin{usecase}
  \ucname{Logueandose en el Sistema}
  \ucpre{True}
  \ucpost{El usuario queda logueado al sistema}
  \ucactor{Usuario}
\end{usecase}
\begin{usecasesteps}
  \ucstep{El usuario ingresa su nombre de usuario y contrase√±a.}{}
  \ucstep{El sistema verifica que los datos sean validos y otorga el acceso.}
        {\ucalt{Si los datos son invalidos, el mostrar el mensaje de datos incorrectos y volver a 1.}}
  \ucstep{Puede ser extendido por \texttt{CU Reservando un viaje} y \texttt{CU Solicitando viaje}}{}
  \ucstep{Fin caso de uso.}{}
\end{usecasesteps}

\begin{usecase}
  \ucname{Efectuando Pago}
  \ucpre{El pago no fue realizado por debito automatico $\land$ El viaje esta en estado finalizado}
  \ucpost{}
  \ucactor{Taxista, Pasajero}
\end{usecase}
\begin{usecasesteps}
  \ucstep{El taxista indica al pasajero el monto a pagar.}{}
  \ucstep{El pasajero paga.}{}
  \ucstep{Si el pago no es exacto, el taxista devuelve vuelto.}{}
  \ucstep{El taxista asenta en el sistema el pago.}{}
  \ucstep{Fin caso de uso.}{}
\end{usecasesteps}
